6/// Extract the remote IP from X-Real-IP header.
7#[tracing::instrument(skip_all)]
8pub async fn extract_real_ip(mut req: Request, next: Next) -> AppMiddlewareResult<Response> {
9    let real_ip: IpAddr = match req.headers().get("X-Real-IP") {
10        Some(header_value) => header_value
11            .to_str()
12            .map_err(|_| AppError::MissingOrInvalidXRealIp)
13            .and_then(|header| {
14                IpAddr::from_str(header).map_err(|_| AppError::MissingOrInvalidXRealIp)
15            })?,
16        None => {
17            #[cfg(debug_assertions)]
18            {
19                use service::testutil::random_ipv6;
20
21                tracing::debug!("no X-Real-IP is given, using random IP address");
22
23                random_ipv6()
24            }
25
26            #[cfg(not(debug_assertions))]
27            return Err(AppError::MissingOrInvalidXRealIp);
28        }
29    };
30
31    req.extensions_mut().insert(real_ip);
32
33    Ok(next.run(req).await)
34}

36/// A key extractor that tries to get rate limiting key from the extension
37/// added by [`extract_real_ip`].
38#[derive(Default, Clone)]
39pub struct GovernorIpKeyExtractor;

41impl KeyExtractor for GovernorIpKeyExtractor {
42    type Key = IpAddr;
43
44    fn extract<T>(&self, req: &Request<T>) -> Result<Self::Key, tower_governor::GovernorError> {
