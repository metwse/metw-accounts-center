3use crate::{

18use std::{net::IpAddr, time::Duration};
19use utoipa::OpenApi;
20
21#[utoipa::path(
22    post, path = "signup/retry",
23    security(("email_verification_session_jwt" = [])),
24    request_body = dto::request::Email,
25    params(dto::request::Captcha),
26    responses(
27        (status = OK)
28    )
29)]
30async fn retry_signup(
31    State(state): State<AppState>,
32    Extension(id): Extension<AccountId>,
33    Extension(real_ip): Extension<IpAddr>,
34    Query(captcha): Query<dto::request::Captcha>,
35    AppJson(email_dto): AppJson<dto::request::Email>,
36) -> AppResult<()> {
37    Ok(AppJson(
38        EmailVerificationSessionHandler(state)
39            .retry_signup(id, email_dto, real_ip, captcha)
40            .await?,
41    ))
42}
43
44pub fn routes(state: AppState) -> Router {
45    Router::new()
46        .route("/signup/retry", post(retry_signup))
47        .layer(limiter::basic::<GovernorIpKeyExtractor>(
48            2,
49            Duration::from_secs(5),
50        ))
51        .route_layer(middleware::from_fn_with_state(
52            state.clone(),
53            auth_email_verification_session,
54        ))
55        .with_state(state)
56}
