1//! # metw-accounts-center app
2//!
3//! This crate contains the application binary, bootstrapping all handlers to
4//! the axum routers.
5//!
6//! *Note*: The Axum application is intended to run under a reverse proxy. The
7//! reverse proxy must provide the remote address via the `X-Real-IP` header.
8
9use axum::{Json, Router, routing::get};
10use serde::Serialize;
11use service::AppState;
12use std::{sync::LazyLock, time::Instant};
13use tower_http::trace::TraceLayer;
14use utoipa::{OpenApi, ToSchema};
15
16/// API routes.
17pub mod routes;
18
19/// API middleware.
20pub mod middleware;
21
22/// API results.
23pub mod res;
24
25/// OpenAPI documentation.
26#[derive(OpenApi)]
27#[openapi(
28    info(description = "metw-accounts-center API"),
29    servers((url = "http://localhost:3781", description = "Local server")),
30    external_docs(url = "https://metwse.github.io/metw-accounts-center/", description = "Crate documentation"),
31    paths(get_status),
32    nest(
33        (path = "/", api = routes::authentication::ApiDoc, tags = ["authentication"]),
34        (path = "/", api = routes::authorization::ApiDoc, tags = ["authorization"]),
35        (path = "/me", api = routes::session::ApiDoc, tags = ["session"]),
36        (path = "/", api = routes::email_verification_session::ApiDoc, tags = ["email_verification_session"])
37    )
38)]
39pub struct ApiDoc;
40
41/// Application status
42#[derive(Serialize, ToSchema)]
43struct AppStatus {
44    /// Status message.
45    pub message: String,
46    /// JavaScript that should be executed by clients if present.
47    pub patch: Option<String>,
48    /// Uptime of the API server, in seconds.
49    pub uptime: u64,
50}
51
52static STARTUP_TIME: LazyLock<Instant> = LazyLock::new(Instant::now);
53
54#[utoipa::path(
55    get, path = "/",
56    responses(
57        (status = OK, body = AppStatus)
58    )
59)]
60async fn get_status() -> Json<AppStatus> {
61    Json(AppStatus {
62        message: "OK".to_string(),
63        patch: None,
64        uptime: Instant::now().duration_since(*STARTUP_TIME).as_secs(),
65    })
66}
67
68/// Constructs the web API.
69pub fn app(state: AppState) -> Router {
70    let _ = *STARTUP_TIME;
71
72    Router::new()
73        .route("/", get(get_status))
74        .route("/openapi.json", get(async || Json(ApiDoc::openapi())))
75        .with_state(state.clone())
76        .merge(routes::authentication::routes(state.clone()))
77        .merge(routes::authorization::routes(state.clone()))
78        .merge(routes::email_verification_session::routes(state.clone()))
79        .merge(routes::session::routes(state))
80        .route_layer(axum::middleware::from_fn(
81            middleware::extract_real_ip::extract_real_ip,
82        ))
83        .layer(TraceLayer::new_for_http())
84}
