25/// Represents the data access layer for account entities.
26///
27/// Entities managed by this repository are [`Account`], [`AccountFlags`],
28/// [`Username`], and [`Email`].
29///
30/// [`Account`]: crate::entity::Account
31/// [`AccountFlags`]: crate::entity::AccountFlags
32/// [`Username`]: crate::entity::Username
33/// [`Email`]: crate::entity::Email
34///
35///
36/// # Invariants
37///
38/// | ID | Constraint | Field |
39/// |--|--|--|
40/// | `I-AR-1` | unique | `accounts.id` |
41/// | `I-AR-2` | unique | `accounts.id, username, username.is_primary == true` |
42/// | `I-AR-3` | unique | `accounts.id, email, email.is_primary == true` |
43/// | `I-AR-4` | check | `username.is_primary == true NAND username.is_expires IS NOT NULL` |
44/// | `I-AR-5` | check | `username == lower(username)` |
45/// | `I-AR-6` | check | `email == lower(email)` |
46/// | `I-AR-7` | unique | `username` |
47/// | `I-AR-8` | unique | `email` |
48///
49/// | ID | Relation | From | To |
50/// |--|--|--|--|
51/// | `R-AR-1`| one-to-many | `account.id` | `email` |
52/// | `R-AR-2`| one-to-many | `account.id` | `username` |
53/// | `R-AR-3`| one-to-exactly one | `account.id` | `account_flags.id` |
54#[async_trait]
55pub trait AccountRepo: Send + Sync {
56    /// Begin a new transactional unit.
57    async fn begin_transaction(&self) -> RepoResult<Box<dyn AccountRepoTransaction>>;
58
59    /// Get password by email.
60    ///
61    /// Returns Argon2-hashed password with account id. Higher layers shall do
62    /// hash verifications.
63    async fn get_login_credentials_by_email(
64        &self,
65        email: &str,
66    ) -> RepoResult<Option<dto::repo::OwnedLoginCredentials>>;
67
68    /// Get password by username.
69    ///
70    /// The behavior is exactly the same with `get_login_with_email`, but also
71    /// includes a flag to check wheter account's email is verified.
72    async fn get_login_credentials_by_username(
73        &self,
74        username: &str,
75    ) -> RepoResult<Option<dto::repo::OwnedLoginCredentials>>;
76
77    /// Get primary username if, exists.
78    ///
79    /// Usually all the accounts have primary usernames, but pending deletion
80    /// might drop usernames of an account.
81    async fn get_primary_username(&self, id: AccountId) -> RepoResult<Option<String>>;
82
83    /// Get usernames by account id.
84    ///
85    /// After username change, previous usernames for will continue to belong
86    /// the account for a while; but they will be garbage collected.
87    async fn get_nonexpiring_username_aliases(&self, id: AccountId) -> RepoResult<Vec<String>>;
88
89    /// Get primary email if, exists.
90    ///
91    /// All regular accounts shall have primary email, but some system accounts
92    /// or deleted accounts do not.
93    async fn get_primary_email(&self, id: AccountId) -> RepoResult<Option<String>>;
94
95    /// Get secondary emails by account id.
96    ///
97    /// Returns an empty list if none have been added.
98    async fn get_secondary_emails(&self, id: AccountId) -> RepoResult<Vec<String>>;
99
100    /// Get account keys - the key bundle of the account.
101    async fn get_keys(&self, id: AccountId) -> RepoResult<Option<dto::repo::OwnedKeys>>;
102
103    /// Set the email primary for the account.
104    ///
105    /// Although `email`s are unique, the `id` parameter is also required to
106    /// prevent race conditions. It is highly unlikely that the owner of an
107    /// email would change at the exact moment their email is being set as
108    /// the primary email, but it is still a safety hazard that must still be
109    /// prevented.
110    async fn set_primary_email_if_current_is(
111        &self,
112        id: AccountId,
113        current_primary_email: &str,
114        new_primary_email: &str,
115    ) -> RepoResult<bool>;
116
117    /// Remove the email if it is not primary email of the account.
118    async fn remove_email_if_not_primary(&self, id: AccountId, email: &str) -> RepoResult<bool>;
119
120    /// Returns true if the username has been taken.
121    async fn is_username_taken(&self, username: &str) -> RepoResult<bool>;
122
123    /// Returns true if the email has been taken.
124    async fn is_email_taken(&self, email: &str) -> RepoResult<bool>;
125
126    /// Returns true if the email has been taken by the given account.
127    async fn is_email_taken_by(&self, id: AccountId, email: &str) -> RepoResult<bool>;
128}

166/// Token revocation state.
167///
168/// A token is considered revoked if any revocation policy invalidates it. This
169/// includes:
170///
171/// - token-level revocation
172/// - scope-level revocation
173/// - account-level revocation
174///
175///
176/// ## Check-and-Revoke Operations
177///
178/// The check-and-revoke methods perform the revocation check and the
179/// corresponding revocation atomically within their own revocation level.
180///
181/// For example, concurrent calls to [`check_and_revoke_token`] guarantee that
182/// at most one call succeeds and all others observe the token as revoked.
183///
184/// However, atomicity is not guaranteed across different revocation levels.
185/// Concurrent calls to [`check_and_revoke_token`] and
186/// [`check_and_revoke_account_tokens_with_scope`] on the same token have
187/// unspecified behavior. Do not use multiple variants of `check_and_revoke*`
188/// for a token scope; use only one level of check-and-revoke.
189///
190/// [`check_and_revoke_token`]: TokenRepo::check_and_revoke_token
191/// [`check_and_revoke_account_tokens_with_scope`]: TokenRepo::check_and_revoke_account_tokens_with_scope
192///
193///
194/// ## Revoke Operations
195///
196/// With these methods, no token revocation check is performed; revocation is
197/// applied directly at the account level.
198#[async_trait]
199pub trait TokenRepo: Send + Sync {
200    /// Check whether the token has been revoked, and if it has not, revoke it.
201    ///
202    /// Returns true if the token has already been revoked.
203    async fn check_and_revoke_token(&self, token: &DecodedToken) -> RepoResult<bool>;
204
205    /// Check whether the token has been revoked, and if it has not, revoke all
206    /// tokens in the account that share the same scope as this token.
207    ///
208    /// Returns true if the token has already been revoked.
209    async fn check_and_revoke_account_tokens_with_scope(
210        &self,
211        token: &DecodedToken,
212    ) -> RepoResult<bool>;
213
214    /// Check whether the token has been revoked, and if it has not, revoke all
215    /// tokens associated with the account.
216    ///
217    /// Returns true if the token has already been revoked.
218    async fn check_and_revoke_account_tokens(&self, token: &DecodedToken) -> RepoResult<bool>;
219
220    /// Revoke all tokens belonging to the account that have the specified
221    /// scope.
222    ///
223    /// This operation atomically updates the revocation cutoff time for the
224    /// `(account_id, scope)` pair and returns the previous cutoff time.
225    async fn revoke_account_tokens_with_scope(
226        &self,
227        account_id: AccountId,
228        scope: &TokenScope,
229    ) -> RepoResult<Option<DateTime<Utc>>>;
230
231    /// Revoke all tokens belonging to the account.
232    ///
233    /// This operation atomically updates the account's token cutoff time,
234    /// and returns the previous cutoff time.
235    async fn revoke_account_tokens(
236        &self,
237        account_id: AccountId,
238    ) -> RepoResult<Option<DateTime<Utc>>>;
239
240    /// Returns true if the token is considered revoked by any revocation rule.
241    async fn is_revoked(&self, token: &DecodedToken) -> RepoResult<bool>;
242}
