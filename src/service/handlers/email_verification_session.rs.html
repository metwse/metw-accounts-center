14/// Account handlers before the email verification.
15///
16/// This handlers *should be* protected using tokens with
17/// [`TokenScope::EmailVerificationSession`]. See [`SessionHandler`] for
18/// details.
19///
20/// [`SessionHandler`]: super::SessionHandler
21pub struct EmailVerificationSessionHandler(pub AppState);
22
23impl EmailVerificationSessionHandler {
24    /// Resends the sign up email.
25    ///
26    /// Sends a [`ConfirmSignup`] email.
27    ///
28    /// [`ConfirmSignup`]: emails::Template::ConfirmSignup
29    #[tracing::instrument(skip(self))]
30    pub async fn retry_signup(
31        self,
32        id: AccountId,
33        email_dto: dto::request::Email,
34        ip: IpAddr,
35        captcha: dto::request::Captcha,
36    ) -> HandlerResult<()> {
37        email_dto.validate()?;
38        if !self.0.captcha_client.validate(captcha.captcha).await {
39            return Err(HandlerError::InvalidCaptcha);
40        }
41
42        let email = email_dto.email;
43
44        if self.0.account_service.is_email_taken(&email).await? {
45            return Err(ServiceError::EmailTaken)?;
46        }
47
48        let Some(username) = self.0.account_service.get_primary_username(id).await? else {
49            return Err(HandlerError::UnexpectedError("account with no username"));
50        };
51
52        self.0
53            .email_limiting_service
54            .check_and_consume_quota(&ip, &email)
55            .await?;
56
57        let complete_signup_jwt = self.0.token_service.sign(&Token {
58            id,
59            scope: TokenScope::CompleteSignup {
60                email: email.clone(),
61            },
62        });
63
64        let template = emails::Template::ConfirmSignup {
65            username,
66            token: complete_signup_jwt,
67        };
68
69        self.0.email_client.send(email, id, template).await;
70
71        Ok(())
72    }
73}
