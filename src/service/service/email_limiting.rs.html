1use crate::{
2    dto,
3    repo::EmailLimitingRepo,
4    service::{ServiceError, ServiceResult},
5};
6use std::net::IpAddr;
7
8/// Email limiting state.
9pub struct EmailLimitingService {
10    repo: Box<dyn EmailLimitingRepo>,
11}
12
13impl EmailLimitingService {
14    /// Creates a new email limiting service.
15    pub fn new(repo: Box<dyn EmailLimitingRepo>) -> Self {
16        Self { repo }
17    }
18
19    /// Same as [`check_and_consume_quota`], but maps `IpLimited` and
20    /// `EmailLimited` to `ServiceResult::EmailLimited`.
21    ///
22    /// [`check_and_consume_quota`]: EmailLimitingRepo::check_and_consume_quota
23    pub async fn check_and_consume_quota(&self, ip: &IpAddr, email: &str) -> ServiceResult<()> {
24        match self.repo.check_and_consume_quota(ip, email).await? {
25            dto::repo::EmailLimitingResult::IpLimited(duration)
26            | dto::repo::EmailLimitingResult::EmailLimited(duration) => {
27                Err(ServiceError::EmailLimited(duration))
28            }
29            dto::repo::EmailLimitingResult::Allowed => Ok(()),
30        }
31    }
32
33    /// See [`refund_ip_quota`].
34    ///
35    /// [`refund_ip_quota`]: EmailLimitingRepo::refund_ip_quota
36    pub async fn refund_ip_quota(&self, ip: &IpAddr, email: &str) -> ServiceResult<()> {
37        Ok(self.repo.refund_ip_quota(ip, email).await?)
38    }
39
40    /// See [`check_and_consume_quota`].
41    ///
42    /// [`check_and_consume_quota`]: EmailLimitingRepo::check_and_consume_quota
43    pub async fn clear_email_limit(&self, email: &str) -> ServiceResult<()> {
44        Ok(self.repo.clear_email_limit(email).await?)
45    }
46}
