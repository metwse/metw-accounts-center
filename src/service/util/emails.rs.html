1use std::{collections::HashMap, sync::LazyLock};
2use strfmt::strfmt;
3use toml;

5/// Email templates.
6///
7/// See [`TokenScope`].
8///
9/// [`TokenScope`]: `crate::token::TokenScope`
10#[allow(missing_docs)]
11#[derive(Clone, Debug)]
12pub enum Template {
13    /// See [`CompleteSignup`].
14    ///
15    /// [`CompleteSignup`]: `crate::token::TokenScope::CompleteSignup`
16    ConfirmSignup { username: String, token: String },
17
18    /// See [`AddEmail`].
19    ///
20    /// [`AddEmail`]: `crate::token::TokenScope::AddEmail`
21    ConfirmNewEmail {
22        username: String,
23        email: String,
24        token: String,
25    },
26
27    /// See [`ChangePrimaryEmail`].
28    ///
29    /// [`ChangePrimaryEmail`]: `crate::token::TokenScope::ChangePrimaryEmail`
30    ConfirmPrimaryEmailChange {
31        username: String,
32        current_primary_email: String,
33        new_primary_email: String,
34        token: String,
35    },
36}
37
38struct ProcessedTemplate {
39    pub subject: String,
40    pub body_text: String,
41    pub body_html: String,
42}
43
44static TEMPLATES: LazyLock<HashMap<String, ProcessedTemplate>> = LazyLock::new(|| {
45    let template_arguments = include_str!("../../email_templates/emails.toml");
46
47    let html_template: &str = include_str!("../../email_templates/template.html");
48    let text_template: &str = include_str!("../../email_templates/template.txt");
49
50    let html_template = String::from_utf8(minify_html::minify(
51        html_template.as_bytes(),
52        &minify_html::Cfg::default(),
53    ))
54    .unwrap();
55
56    let templates =
57        toml::from_str::<HashMap<String, HashMap<String, String>>>(template_arguments).unwrap();
58
59    let mut processed_templates = HashMap::<String, ProcessedTemplate>::new();
60
61    for (template_name, mut value) in templates.into_iter() {
62        let body_html = strfmt(&html_template, &value).unwrap();
63        let body_text = strfmt(text_template, &value).unwrap();
64
65        processed_templates.insert(
66            template_name,
67            ProcessedTemplate {
68                subject: value.remove("subject").unwrap(),
69                body_text,
70                body_html,
71            },
72        );
73    }
74
75    processed_templates
76});
77
78impl Template {
79    /// Get subject of the template.
80    pub fn subject(&self) -> String {
81        TEMPLATES.get(self.to_str()).unwrap().subject.clone()
82    }
83
84    /// Get email body of the template.
85    pub fn body_html(&self, callback_url: &str) -> String {
86        let template = &TEMPLATES.get(self.to_str()).unwrap().body_html;
87
88        self.build_email(template, callback_url)
89    }
90
91    /// Get plaintext email body of the template.
92    pub fn body_text(&self, callback_url: &str) -> String {
93        let template = &TEMPLATES.get(self.to_str()).unwrap().body_text;
94
95        self.build_email(template, callback_url)
96    }
97
98    fn build_email(&self, template: &str, callback_url: &str) -> String {
99        match self {
100            Self::ConfirmSignup { username, token } => strfmt!(
101                template,
102                callback_url => callback_url.to_string(),
103                token => token.to_string(),
104                username => username.to_string()
105            )
106            .unwrap(),
107            Self::ConfirmNewEmail {
108                username,
109                email,
110                token,
111            } => strfmt!(
112                template,
113                callback_url => callback_url.to_string(),
114                token => token.to_string(),
115                username => username.to_string(),
116                email => email.to_string()
117            )
118            .unwrap(),
119            Self::ConfirmPrimaryEmailChange {
120                username,
121                current_primary_email,
122                new_primary_email,
123                token,
124            } => strfmt!(
125                template,
126                callback_url => callback_url.to_string(),
127                token => token.to_string(),
128                username => username.to_string(),
129                current_primary_email => current_primary_email.to_string(),
130                new_primary_email => new_primary_email.to_string()
131            )
132            .unwrap(),
133        }
134    }
135
136    fn to_str(&self) -> &'static str {
137        match self {
138            Self::ConfirmSignup { .. } => "confirm-signup",
139            Self::ConfirmNewEmail { .. } => "confirm-new-email",
140            Self::ConfirmPrimaryEmailChange { .. } => "confirm-primary-email-change",
141        }
142    }
143}
